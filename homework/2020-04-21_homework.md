### ![class](https://github.com/tootoomaa/FastCampus_IOS_school/blob/master/homework/class.png)

 ```swift
import Foundation

class Animal {
    var brain:Bool = true
    var legs:Int = 0
}

class Human: Animal {
    override var legs:Int {
        get { super.legs}
        set { super.legs = newValue }
    }
}

let animal = Animal()
animal.brain		// true
animal.legs			// 0
let human = Human()
human.legs = 2		
human.legs			// 2
animal.legs			// 0

class Pet: Animal {
    var fleas:Int = 0
    override var legs: Int {
        get { super.legs }
        set { super.legs = newValue }
    }
}

class Dog :Pet {
    override var fleas: Int {
        get { super.fleas }
        set { super.fleas = newValue}
    }
}

class Cat :Pet {
    override var fleas: Int {
        get { super.fleas }
        set { super.fleas = newValue}
    }
}

let dog = Dog()
let cat = Cat()
let pet = Pet()

pet.legs = 4
pet.legs			// 4

dog.fleas = 8
dog.fleas			// 8
cat.fleas = 4
cat.fleas			// 4
 ```
